Quick Example

// Install via NuGet
dotnet add package Aspekt

// Create your aspect
public class LoggingAspect : Aspect
{
    public override void OnEntry(MethodArguments args)
        => Console.WriteLine($"Entering: {args.Name}");
}

// Apply it to any method
[LoggingAspect]
public void ProcessData() 
{ 
    // Your code here
}
